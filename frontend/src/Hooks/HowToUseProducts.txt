import { useProducts } from '../hooks/useProducts';

function ProductsPage() {
  const {
    products,
    pagination,
    loading,
    setPage,
    setSearch
  } = useProducts();

  return (
    <div>
      <input 
        onChange={(e) => setSearch(e.target.value)}
        placeholder="Search products..."
      />
      
      {loading ? (
        <Spinner />
      ) : (
        <ProductList products={products} />
      )}

      <Pagination 
        current={pagination.page}
        total={pagination.pages}
        onChange={setPage}
      />
    </div>
  );
}
